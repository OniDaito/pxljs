MouseEmitter["_setMouseEvent"] = (event) ->
  @_getMousePos event
  @_setButtons event

  event.preventDefault() if event.preventDefault?
  event.stopImmediatePropagation() if event.stopImmediatePropagation?
  event.stopPropagation() if event.stopPropagation?
  event
