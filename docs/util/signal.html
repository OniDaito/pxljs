      del : (func, context) ->
    for obj in @listeners
      if obj.c == context
        if obj.f == func
          i = @listeners.indexOf(obj) 
          @listeners.splice(i, 1)  
          break

    if @_pause
      return @

    removals = []
    for l in @listeners
 
      PXL.Context.switchContext l.g

      l.f.apply(l.c, arguments)
      if l.o
        removals.push l
    
    for l in removals
      @del l
